133 		return o !== undefined && o !== null ? (o instanceof Array && !o.callee ? o : (typeof o.length !== "number" || o.split || o.setInterval || o.call ? [ o ] : Array.prototype.slice.call(o))) : [];

136 	function keys(o) {
137 		var result = [], key;
138 		
139 		switch (typeOf(o)) {
140 		case "object":
141 			for (key in o) {
142 				if (o[key] !== O[key]) {
143 					result[result.length] = key;
144 				}
145 			}
146 			break;
147 		case "array":
148 			for (key = o.length - 1; key >= 0; --key) {
149 				result[key] = key;
150 			}
151 			break;
152 		}
153 		
154 		return result;
155 	}

165 	//Produces a JSV acceptable clone
166 	function clone(obj, deep) {
167 		var newObj, x;
168 		
169 		switch (typeOf(obj)) {
170 		case "object":
171 			if (deep) {
172 				newObj = {};
173 				for (x in obj) {
174 					if (obj[x] !== O[x]) {
175 						newObj[x] = clone(obj[x], deep);
176 					}
177 				}
178 				return newObj;
179 			} else {
180 				return createObject(obj);
181 			}
182 			break;
183 		case "array":
184 			if (deep) {
185 				newObj = new Array(obj.length);
186 				x = obj.length;
187 				while (--x >= 0) {
188 					newObj[x] = clone(obj[x], deep);
189 				}
190 				return newObj;
191 			} else {
192 				return Array.pototype.slice.call(obj);
193 			}
194 			break;
195 		default:
196 			return obj;
197 		}
198 	}

251 	//merge two objects together that is (roughly) extends compliant
252 	function mergeSchemas(base, extra, isSchema) {
253 		var key;
254 		for (key in extra) {
255 			if (extra[key] !== O[key]) {
256 				if (isSchema && key === "extends") {
257 					base[key] = toArray(base[key]).concat(toArray(extra[key]));
258 				} else if (typeOf(base[key]) === "object" && typeOf(extra[key]) === "object") {
259 					mergeSchemas(base[key], extra[key], !isSchema || key !== "properties");  //FIXME: An attribute other then "properties" may be a plain object
260 				} else {
261 					base[key] = extra[key];
262 				}
263 			}
264 		}
265 		return base;
266 	}

494 	function JSONSchema(env, json, uri, schema) {
495 		var fr;
496 		JSONInstance.call(this, env, json, uri);
497 		
498 		if (schema === true) {
499 			this._schema = this;
500 		} else if (json instanceof JSONSchema && !(schema instanceof JSONSchema)) {
501 			this._schema = json._schema;  //TODO: Make sure cross environments don't mess everything up
502 		} else {
503 			this._schema = schema instanceof JSONSchema ? schema : this._env.getDefaultSchema() || JSONSchema.createEmptySchema(this._env);
504 		}
505 		
506 		//determine fragment delimiter from schema
507 		fr = this._schema.getValueOfProperty("fragmentResolution");
508 		if (fr === "dot-delimited") {
509 			this._fd = ".";
510 		} else if (fr === "slash-delimited") {
511 			this._fd = "/";
512 		}
513 	}

579 	JSONSchema.prototype.validate = function (instance, report, parent, parentSchema) {
580 		var validator = this._schema.getValueOfProperty("validator");
581 		
582 		if (!(instance instanceof JSONInstance)) {
583 			instance = this.getEnvironment().createInstance(instance);
584 		}
585 		
586 		if (!(report instanceof Report)) {
587 			report = new Report();
588 		}
589 		
590 		if (typeof validator === "function" && !report.isValidatedBy(instance.getURI(), this.getURI())) {
591 			report.registerValidation(instance.getURI(), this.getURI());
592 			validator(instance, this, this._schema, report, parent, parentSchema);
593 		}
594 		
595 		return report;
596 	};

642 	/**
643 	 * @param data
644 	 * @param {JSONSchema} schema
645 	 * @param {String} uri
646 	 * @returns {JSONSchema}
647 	 */
648 	
649 	Environment.prototype.createSchema = function (data, schema, uri) {
650 		var instance, 
651 			initializer,
652 			instanceProperties,
653 			schemaProperties,
654 			schemaProperty,
655 			parser,
656 			key;
657 		uri = formatURI(uri);
658 		
659 		if (data instanceof JSONSchema && (!uri || data._uri === uri) && (!schema || data._schema.equals(schema))) {
660 			return data;
661 		}
662 		
663 		instance = new JSONSchema(this, data, uri, schema);
664 		
665 		initializer = instance.getSchema().getValueOfProperty("initializer");
666 		if (typeof initializer === "function") {
667 			instance = initializer(instance);
668 		}
669 		
670 		//register schema
671 		this._schemas[instance._uri] = instance;
672 		
673 		//build & cache the rest of the schema
674 		if (!instance._attributes && instance.getType() === "object") {
675 			instanceProperties = instance.getProperties();
676 			schemaProperties = instance._schema.getProperty("properties");
677 			instance._attributes = {};
678 			for (key in instanceProperties) {
679 				if (instanceProperties[key] !== O[key]) {
680 					schemaProperty = schemaProperties && schemaProperties.getProperty(key);
681 					parser = schemaProperty && schemaProperty.getValueOfProperty("parser");
682 					if (typeof parser === "function") {
683 						instance._attributes[key] = parser(instanceProperties[key], schemaProperty);
684 					} else {
685 						instance._attributes[key] = instanceProperties[key].getValue();
686 					}
687 				}
688 			}
689 		}
690 		
691 		return instance;
692 	};

748 	 * @param instanceJSON
749 	 * @param schemaJSON
750 	 * @returns {Report}
751 	 */
752 	
753 	Environment.prototype.validate = function (instanceJSON, schemaJSON) {
754 		var instance = this.createInstance(instanceJSON),
755 			schema = this.createSchema(schemaJSON),
756 			schemaSchema = schema.getSchema(),
757 			report = new Report();
758 		
759 		report.instance = instance;
760 		report.schema = schema;
761 		report.schemaSchema = schemaSchema;
762 		
763 		if (schemaJSON && !(schemaJSON instanceof JSONSchema)) {
764 			schemaSchema.validate(schema, report);
765 			
766 			if (report.errors.length) {
767 				return report;
768 			}
769 		}
770 		
771 		return schema.validate(instance, report);
772 	};
